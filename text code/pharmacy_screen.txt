    required String buttonText,

    return Container(
      padding: const EdgeInsets.all(18),
      decoration: BoxDecoration(
        color: Colors.white,
        borderRadius: BorderRadius.circular(20),

          const SizedBox(width: 15),
          Expanded(

              crossAxisAlignment: CrossAxisAlignment.start,
              children: [

                    fontSize: 16,

                ElevatedButton(
